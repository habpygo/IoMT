package main

import (
	"encoding/json"
	"fmt"
	"os"

	"crypto/sha256"
)

func CheckSum(bs []byte) []byte {
	h := sha256.New()
	h.Write(bs)
	return h.Sum(nil)
}

func CheckStringSum(bs string) []byte {
	h := sha256.New()
	h.Write([]byte(bs))
	return h.Sum(nil)
}

func main() {
	type ColorGroup struct {
		ID     int
		Name   string
		Colors []string
	}
	group := ColorGroup{
		ID:     1,
		Name:   "Reds",
		Colors: []string{"Crimson", "Red", "Ruby", "Maroon"},
	}
	b, err := json.Marshal(group)
	if err != nil {
		fmt.Println("error:", err)
	}
	os.Stdout.Write(b)
	fmt.Println("\nafter Marshalling we get: ", b)
	fmt.Println("\nafter Marshalling and stringifying we get: ", string(b))
	merkle := CheckSum(b)
	stringMerkle := CheckStringSum(string(b))
	fmt.Println("CheckSum of byte slice b is: ", merkle)
	fmt.Println("CheckSum of string(b) is ", stringMerkle)
}
